短路逻辑：||= &&= ??= 三元运算符杀手

// 老写法
config.timeout = config.timeout || 5000

// 新语法糖
config.timeout ||= 5000

// 只补 null/undefined
config.retry ??= 3

?? 与 || 区别：前者不拦截 0 与 ''，做配置默认值时更安全。

8. Promise.allSettled 批量请求不中断

const reqs = [api.getA(), api.getB(), api.getC()]
const res = await Promise.allSettled(reqs)

const succ = res
  .filter(r => r.status === 'fulfilled')
  .map(r => r.value)
const fail = res
  .filter(r => r.status === 'rejected')
  .map(r => r.reason)

节流（throttle）与防抖（debounce），好像要背手写一遍

// 防抖：连续触发只执行最后一次
function debounce(fn, delay = 300) {
  let timer = null
  return function (...args) {
    clearTimeout(timer)
    timer = setTimeout(() => fn.apply(this, args), delay)
  }
}
