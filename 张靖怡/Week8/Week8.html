节流（throttle）与防抖（debounce） 板

// 防抖：连续触发只执行最后一次
function debounce(fn, delay = 300) {
  let timer = null
  return function (...args) {
    clearTimeout(timer)
    timer = setTimeout(() => fn.apply(this, args), delay)
  }
}

// 节流：单位时间内只执行一次
function throttle(fn, wait = 300) {
  let prev = 0
  return function (...args) {
    const now = Date.now()
    if (now - prev > wait) {
      prev = now
      fn.apply(this, args)
    }
  }
}

练习：给搜索框加 debounce，给滚动加载加 throttle，console 里看调用次数立刻降 90%。

深拷贝：JSON 的坑 & 简易递归实现

const obj = { a: 1, b: { c: 2 }, d: new Date() }

// JSON 版（函数、Date、undefined 会丢）
const bad = JSON.parse(JSON.stringify(obj))

// 递归版（只处理普通对象/数组）
function deepClone(target) {
  if (typeof target !== 'object' || target === null) return target
  const clone = Array.isArray(target) ? [] : {}
  for (const key in target) {
    clone[key] = deepClone(target[key])
  }
  return clone
}

模块化：import / export 花式写法

// 一条 export
export const api = axios.create({})

// 默认导出
export default api

// 动态导入（按需加载）
const LazyComp = () => import('./LazyComp.vue')

路由懒加载、组件异步注册都是动态 import，返回 Promise。

安全：XSS 与 CSP 基础

  • 永远不要 `innerHTML = 用户输入`
  • 用 `textContent` 或 Vue 的 `{{ }}`（已自动转义）
  • 后端返回 HTML 片段时加 CSP 响应头： Content-Security-Policy: default-src 'self'

在 axios 拦截器里统一给 Cookie 加 SameSite=Strict，防 CSRF。
